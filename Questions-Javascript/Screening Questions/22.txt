function checkNumberPowerOfTen(number){
  if(number>=10){
    return (checkNumberPowerOfTen(number/10));
  } 
  else if(number<=0.1) {
    return (checkNumberPowerOfTen(number*10));
  }
  else if(number === 1) {
    return true
  }
  else {
    return false;
  }
}
